43  double KNNClassifier<T, Callable>::evaluate(const Point<T> &p, bool raw_value) {
44  assert(this->samples->dim() == p.size());
45  auto points = this->samples->points();
46  std::vector<double> distances(this->samples->size());
47  std::vector<int> classes = this->samples->classes();
48  std::vector<size_t> idx(distances.size());
49  std::vector<PointPointer<T>> neigh;
50  auto p0 = std::make_shared<Point<T> >(p);
51 
52  if(algorithm == "brute"){
53  // fill the index vector
54  std::iota(idx.begin(), idx.end(), 0);
55 
56  // compute the metrics from the sample to be evaluated to the samples vector
57  std::transform(points.begin(), points.end(), distances.begin(),
58  [&p0, this](const std::shared_ptr<Point<T> > q) {
59  return this->dist_function(*p0, *q);
60  });
61 
62  // sort the index vector by the metrics from the sample to be evaluated
63  std::nth_element(idx.begin(), idx.begin() + this->k, idx.end(), [&distances](size_t i1, size_t i2) {
64  return distances[i1] < distances[i2];
65  });
66  }
67  // find the most frequent class in the k nearest neighbors
68  size_t max_index = 0, max_freq = 0, i=0;
69  double s = 0.0001, max_prob = 0.0;
70  std::for_each(classes.begin(), classes.end(), [&idx, &points, &i, &max_freq, &max_index, &max_prob, &s, &classes,&neigh, this](int c){
71  int freq = 0;
72  if(algorithm == "brute") {
73  freq = std::count_if(idx.begin(), idx.begin() + this->k, [&points, &c](size_t id) {
74  return points[id]->Y() == c;
75  });
76  }else{
77  freq = std::count_if(neigh.begin(), neigh.end(), [&c](auto point) {
78  return point->Y() == c;
79  });
80  }
81 
82  if(freq > max_freq){
83  double prob = (freq+s)/(k+classes.size()*s);
84  max_index = i;
85  max_freq = freq;
86  max_prob = prob;
87  }
88  i++;
89  });
90  this->pred_prob = max_prob;
91  return classes[max_index];
92  }
