220  double Kernel<T>::function(std::shared_ptr<Point< T > > one, std::shared_ptr<Point< T > > two, int dim,
221  typename FunctionType<T>::Type f) const{
222  int i = 0;
223  double t, sum = 0.0;
224  std::vector< T > a = one->X(), b = two->X();
225 
226  // a.erase(a.end());
227  //b.erase(b.end());
228 
229  switch(type)
230  {
231  case mltk::INNER_PRODUCT: //Produto Interno
232  for(i = 0; i < dim; ++i)
233  sum += a[i] * b[i];
234  break;
235  case mltk::POLYNOMIAL: //Polinomial
236  for(i = 0; i < dim; ++i)
237  sum += a[i] * b[i];
238  // sum = (param > 1) ? std::pow(sum+1, param) : sum;
239  sum = (param > 1) ? std::pow(sum, param) : sum;
240  break;
241 
242  case mltk::GAUSSIAN: //Gaussiano
243  for(i = 0; i < dim; ++i)
244  { t = a[i] - b[i]; sum += t * t; }
245  sum = std::exp(-1 * sum * param);
246  break;
247 
248  case mltk::CUSTOM:
249  if(f) sum = f(*one, *two, param);
250  else sum = this->func(*one, *two, param);
251  break;
252  }
253 
254  /*The '+1' here accounts for the bias term "b" in SVM formulation since
255  <w,x> = \sum_i \alpha_i y_i k(x_i,x) + b and b=\sum_i \alpha_i y_i*/
256 
257  return sum;// + 1.0f;
